## 代码
### C++ Code:
```cpp
vector<string> letterCombinations(string digits) {
    //map映射字母和数字
    unordered_map<char,vector<string>> phone;
    phone['2']={"a","b","c"};
    phone['3']={"d","e","f"};
    phone['4']={"g","h","i"};
    phone['5']={"j","k","l"};
    phone['6']={"m","n","o"};
    phone['7']={"p","q","r","s"};
    phone['8']={"t","u","v"};
    phone['9']={"w","x","y","z"};

    vector<string>ans;
    for(auto i:digits )
    {   //对每个数字循环
        if(ans.empty())
        {
            ans=phone[i];//第一个字母直接全部push
            continue;
         }
        //每次取出当前ans中的全部组合
        vector<string> temp=ans;
        ans.clear();
        //并清空此时的ans
        for(auto j:temp)
        {
            //对于当前字母一一与当前ans中的字符串组合
            for (auto k: phone[i])
            {
                ans.push_back(j+k);//每个组合都加入ans
            }
        }
    }
    return ans;
}
```
